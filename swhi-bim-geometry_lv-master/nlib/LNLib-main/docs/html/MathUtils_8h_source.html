61 {
62 std::vector<T> temp;
63
64 for (int i = 0; i < matrix[0].size(); i++)
65 {
66 for (int j = 0; j < matrix.size(); j++)
67 {
68 temp.emplace_back(matrix[j][i]);
69 }
70 transposed.emplace_back(temp);

72 }
