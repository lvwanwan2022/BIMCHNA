76Matrix4d LNLib::Matrix4d::CreateReflection(const XYZ& normal, double distanceFromOrigin)
77{
78 Matrix4d result = Matrix4d();
79
80 XYZ t = const_cast<XYZ&>(normal).Normalize();
81 double x = t[0];
82 double y = t[1];
83 double z = t[2];
84
85 result.m_matrix4d[0][0] = -2 * x * x + 1;
86 result.m_matrix4d[0][1] = -2 * y * x;
87 result.m_matrix4d[0][2] = -2 * z * x;
88 result.m_matrix4d[0][3] = -2 * x * distanceFromOrigin;
89
90 result.m_matrix4d[1][0] = -2 * x * y;
91 result.m_matrix4d[1][1] = -2 * y * y + 1;
92 result.m_matrix4d[1][2] = -2 * z * y;
93 result.m_matrix4d[1][3] = -2 * y * distanceFromOrigin;
94
95 result.m_matrix4d[2][0] = -2 * x * z;
96 result.m_matrix4d[2][1] = -2 * y * z;
97 result.m_matrix4d[2][2] = -2 * z * z + 1;
98 result.m_matrix4d[2][3] = -2 * z * distanceFromOrigin;
99
100 result.m_matrix4d[3][0] = 0.0;
101 result.m_matrix4d[3][1] = 0.0;
102 result.m_matrix4d[3][2] = 0.0;
103 result.m_matrix4d[3][3] = 1.0;
104 return result;
105}

112Matrix4d LNLib::Matrix4d::CreateRotation(const XYZ& axis, double rad)
113{
114 double c = cos(rad);
115 double s = sin(rad);
116 double t = 1.0 - c;
117
118 XYZ nAxis = const_cast<XYZ&>(axis).Normalize();
119 double x = nAxis.GetX();
120 double y = nAxis.GetY();
121 double z = nAxis.GetZ();
122
123 Matrix4d result = Matrix4d();
124 result.m_matrix4d[0][0] = c + x * x * t;
125 result.m_matrix4d[0][1] = x * y * t - z * s;
126 result.m_matrix4d[0][2] = x * z * t + y * s;
127 result.m_matrix4d[0][3] = 0.0;
128
129 result.m_matrix4d[1][0] = y * x * t + z * s;
130 result.m_matrix4d[1][1] = c + y * y * t;
131 result.m_matrix4d[1][2] = y * z * t - x * s;
132 result.m_matrix4d[1][3] = 0.0;
133
134 result.m_matrix4d[2][0] = z * x * t - y * s;
135 result.m_matrix4d[2][1] = z * y * t + x * s;
136 result.m_matrix4d[2][2] = c + z * z * t;
137 result.m_matrix4d[2][3] = 0.0;
138
139 result.m_matrix4d[3][0] = 0.0;
140 result.m_matrix4d[3][1] = 0.0;
141 result.m_matrix4d[3][2] = 0.0;
142 result.m_matrix4d[3][3] = 1.0;
143 return result;
144}

146Matrix4d LNLib::Matrix4d::CreateRotationAtPoint(const XYZ& origin, const XYZ& axis, double rad)
147{
148 Matrix4d rodrigues = CreateRotation(axis, rad);
149 double x = rodrigues.GetElement(0, 0) * origin[0] + rodrigues.GetElement(0, 1) * origin[1] + rodrigues.GetElement(0, 2) * origin[2];
150 double y = rodrigues.GetElement(1, 0) * origin[0] + rodrigues.GetElement(1, 1) * origin[1] + rodrigues.GetElement(1, 2) * origin[2];
151 double z = rodrigues.GetElement(2, 0) * origin[0] + rodrigues.GetElement(2, 1) * origin[1] + rodrigues.GetElement(2, 2) * origin[2];
152 XYZ Rv = XYZ(x,y,z);
153 XYZ offset = origin - Rv;
154 rodrigues.SetBasisW(offset);
155 return rodrigues;
156}

261XYZ LNLib::Matrix4d::OfPoint(const XYZ& point)
262{
263 double x = m_matrix4d[0][0] * point[0] + m_matrix4d[0][1] * point[1] + m_matrix4d[0][2] * point[2] + m_matrix4d[0][3] * 1;
264 double y = m_matrix4d[1][0] * point[0] + m_matrix4d[1][1] * point[1] + m_matrix4d[1][2] * point[2] + m_matrix4d[1][3] * 1;
265 double z = m_matrix4d[2][0] * point[0] + m_matrix4d[2][1] * point[1] + m_matrix4d[2][2] * point[2] + m_matrix4d[2][3] * 1;
266
267 double w = m_matrix4d[3][0] * point[0] + m_matrix4d[3][1] * point[1] + m_matrix4d[3][2] * point[2] + m_matrix4d[3][3] * 1;
268
269 return XYZ(x,y,z)/w;

295bool LNLib::Matrix4d::GetInverse(Matrix4d& inverse)
296{
297 int n = 4;
298 std::vector<std::vector<double>> currentMatrix(n);
299 for (int i = 0; i < n; i++)
300 {
301 currentMatrix[i].resize(n);
302 for (int j = 0; j < n; j++)
303 {
304 currentMatrix[i][j] = m_matrix4d[i][j];
305 }
306 }
307 std::vector<std::vector<double>> inverseMatrix;
308 if (!MathUtils::MakeInverse(currentMatrix, inverseMatrix))
309 {
310 return false;
311 }
312 for (int i = 0; i < n; i++)
313 {
314 for (int j = 0; j < n; j++)
315 {
316 inverse.m_matrix4d[i][j] = inverseMatrix[i][j];
317 }
318 }
319 return true;
320}

348bool LNLib::Matrix4d::IsIdentity()
349{
350 bool c1 = MathUtils::IsAlmostEqualTo(m_matrix4d[0][0], 1);
351 if (!c1)
352 {
353 return false;
354 }
355 bool c2 = MathUtils::IsAlmostEqualTo(m_matrix4d[1][1], 1);
356 if (!c2)
357 {
358 return false;
359 }
360 bool c3 = MathUtils::IsAlmostEqualTo(m_matrix4d[2][2], 1);
361 if (!c3)
362 {
363 return false;
364 }
365
366 double sum = m_matrix4d[0][1] + m_matrix4d[0][2] + m_matrix4d[0][3] +
367 m_matrix4d[1][0] + m_matrix4d[1][2] + m_matrix4d[1][3] +
368 m_matrix4d[2][0] + m_matrix4d[2][1] + m_matrix4d[2][3] +
369 m_matrix4d[3][0] + m_matrix4d[3][1] + m_matrix4d[3][2];
370 bool c4 = MathUtils::IsAlmostEqualTo(sum, 0.0);
371 if (!c4)
372 {
373 return false;
374 }
