18CurveCurveIntersectionType Intersection::ComputeRays(const XYZ& point0, const XYZ& vector0, const XYZ& point1, const XYZ& vector1, double& param0, double& param1, XYZ& intersectPoint)
19{
20 VALIDATE_ARGUMENT(!vector0.IsZero(), "vector0", "Vector0 must not be zero vector.");
21 VALIDATE_ARGUMENT(!vector1.IsZero(), "vector1", "Vector1 must not be zero vector.")
22
23 XYZ v0 = const_cast<XYZ&>(vector0);
24 XYZ v1 = const_cast<XYZ&>(vector1);
25
26 XYZ cross = v0.CrossProduct(v1);
27 XYZ diff = point1 - point0;
28 XYZ coinCross = diff.CrossProduct(v1);
29
30 if (cross.IsZero())
31 {
32 if (coinCross.IsZero())
33 {
34 return CurveCurveIntersectionType::Coincident;
35 }
36 else
37 {
38 return CurveCurveIntersectionType::Parallel;
39 }
40 }
41
42 double crossLength = cross.Length();
43 double squareLength = crossLength * crossLength;
44
45 XYZ pd1Cross = diff.CrossProduct(vector1);
46 double pd1Dot = pd1Cross.DotProduct(cross);
47 param0 = pd1Dot / squareLength;
48
49 XYZ pd2Cross = diff.CrossProduct(vector0);
50 double pd2Dot = pd2Cross.DotProduct(cross);
51 param1 = pd2Dot / squareLength;
52
53 XYZ rayP0 = point0 + vector0 * param0;
54 XYZ rayP1 = point1 + vector1 * param1;
55
56 if (rayP0.IsAlmostEqualTo(rayP1))
57 {
58 intersectPoint = rayP0;
59 return CurveCurveIntersectionType::Intersecting;
60 }
61 return CurveCurveIntersectionType::Skew;
62}

64LinePlaneIntersectionType LNLib::Intersection::ComputeLineAndPlane(const XYZ& normal, const XYZ& pointOnPlane, const XYZ& pointOnLine, const XYZ& lineDirection, XYZ& intersectPoint)
65{
66 XYZ planeNormal = const_cast<XYZ&>(normal).Normalize();
67 XYZ lineDirectionNormal = const_cast<XYZ&>(lineDirection).Normalize();
68 XYZ P2L = pointOnLine - pointOnPlane;
69 XYZ P2LNormal = P2L.Normalize();
70 double dot = P2LNormal.DotProduct(planeNormal);
71 if (MathUtils::IsAlmostEqualTo(dot, 0.0))
72 {
73 return LinePlaneIntersectionType::On;
74 }
75 double angle = planeNormal.AngleTo(lineDirectionNormal);
76 if (MathUtils::IsAlmostEqualTo(angle, Constants::Pi / 2))
77 {
78 return LinePlaneIntersectionType::Parallel;
79 }
80 double d = -P2L.DotProduct(planeNormal) / lineDirection.DotProduct(planeNormal);
81 intersectPoint = d * lineDirectionNormal + pointOnLine;
82 return LinePlaneIntersectionType::Intersecting;
83}
