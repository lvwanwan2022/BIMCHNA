100UV UV::Normalize()
101{
102 double length = Length();
103 UV newUV = *this;
104 if (length > 0)
105 {
106 double invLength = (double)(1.0 / length);
107 newUV.m_uv[0] *= invLength;
108 newUV.m_uv[1] *= invLength;
109 }
110 return newUV;
111}
