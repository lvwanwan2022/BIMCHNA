19 double Integrator::Simpson(IntegrationFunction& function, void* customData, double start, double end)
20 {
21 double st = (function)(start, customData);
22 double mt = (function)((start + end) / 2.0, customData);
23 double et = (function)((end), customData);
24 double result = ((end - start) / 6.0) * (st + 4 * mt + et);
25 return result;
26 }

28 double Integrator::Simpson(BinaryIntegrationFunction& function, void* customData, double uStart, double uEnd, double vStart, double vEnd)
29 {
30 double du = uEnd - uStart;
31 double dv = vEnd - vStart;
32 double hdu = 0.5 * du;
33 double hdv = 0.5 * dv;
34
35 // Sample 9 points with weights
36 int sampleNumber = 9;
37 int patches = 4;
38 double uvw[27] = {
39 uStart, vStart, 1,
40 uStart, vStart + hdv, 4,
41 uStart, vEnd, 1,
42 uStart + hdu, vStart, 4,
43 uStart + hdu, vStart + hdv, 16,
44 uStart + hdu, vEnd, 4,
45 uEnd, vStart, 1,
46 uEnd, vStart + hdv, 4,
47 uEnd, vEnd, 1,
48 };
49
50 double sum = 0;
51 for(int i=0; i < sampleNumber; ++i)
52 {
53 double* base = uvw + i*3;
54 double u = base[0];
55 double v = base[1];
56 double w = base[2];
57 double f = function(u, v, customData);
58 sum += w * f;
59 }
60
61 sum *= du * dv / (sampleNumber * patches);
62 return sum;
63 }

121 std::vector<double> Integrator::ChebyshevSeries(int size)
122 {
123 std::vector<double> series(size);
124
125 int lenw = series.size() - 1;
126 int j, k, l, m;
127 double cos2, sin1, sin2, hl;
128
129 cos2 = 0;
130 sin1 = 1;
131 sin2 = 1;
132 hl = 0.5;
133 k = lenw;
134 l = 2;
135 while (l < k - l - 1)
136 {
137 series[0] = hl * 0.5;
138 for (j = 1; j <= l; j++)
139 {
140 series[j] = hl / (1 - 4 * j * j);
141 }
142 series[l] *= 0.5;
143 dfct(l, 0.5 * cos2, sin1, series);
144 cos2 = std::sqrt(2 + cos2);
145 sin1 /= cos2;
146 sin2 /= 2 + cos2;
147 series[k] = sin2;
148 series[k - 1] = series[0];
149 series[k - 2] = series[l];
150 k -= 3;
151 m = l;
152 while (m > 1)
153 {
154 m >>= 1;
155 for (j = m; j <= l - m; j += (m << 1))
156 {
157 series[k] = series[j];
158 k--;
159 }
160 }
161 hl *= 0.5;
162 l *= 2;
163 }
164 return series;
165 }
166 double Integrator::ClenshawCurtisQuadrature(IntegrationFunction& function, void* customData, double start, double end, std::vector<double>& series, double epsilon)
167 {
168 double integration;
169 int j, k, l;
170 double err, esf, eref, erefh, hh, ir, iback, irback, ba, ss, x, y, fx, errir;
171 int lenw = series.size() - 1;
172 esf = 10;
173 ba = 0.5 * (end - start);
174 ss = 2 * series[lenw];
175 x = ba * series[lenw];
176 series[0] = 0.5 * (function)(start, customData);
177 series[3] = 0.5 * (function)(end, customData);
178 series[2] = (function)(start + x, customData);
179 series[4] = (function)(end - x, customData);
180 series[1] = (function)(start + ba, customData);
181 eref = 0.5 * (fabs(series[0]) + std::fabs(series[1]) + std::fabs(series[2]) + std::fabs(series[3]) + std::fabs(series[4]));
182 series[0] += series[3];
183 series[2] += series[4];
184 ir = series[0] + series[1] + series[2];
185 integration = series[0] * series[lenw - 1] + series[1] * series[lenw - 2] + series[2] * series[lenw - 3];
186 erefh = eref * std::sqrt(epsilon);
187 eref *= epsilon;
188 hh = 0.25;
189 l = 2;
190 k = lenw - 5;
191 do {
192 iback = integration;
193 irback = ir;
194 x = ba * series[k + 1];
195 y = 0;
196 integration = series[0] * series[k];
197 for (j = 1; j <= l; j++) {
198 x += y;
199 y += ss * (ba - x);
200 fx = (function)(start + x, customData) + (function)(end - x, customData);
201 ir += fx;
202 integration += series[j] * series[k - j] + fx * series[k - j - l];
203 series[j + l] = fx;
204 }
205 ss = 2 * series[k + 1];
206 err = esf * l * std::fabs(integration - iback);
207 hh *= 0.25;
208 errir = hh * std::fabs(ir - 2 * irback);
209 l *= 2;
210 k -= l + 2;
211 } while ((err > erefh || errir > eref) && k > 4 * l);
212 integration *= end - start;
213 if (err > erefh || errir > eref)
214 {
215 err *= -fabs(end - start);
216 }
217 else
218 {
219 err = eref * std::fabs(end - start);
220 }
221 return integration;
