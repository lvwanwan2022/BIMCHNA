25XYZ LNLib::BezierCurve::GetPointOnQuadraticArc(const XYZW& startPoint, const XYZW& middlePoint, const XYZW& endPoint, double paramT)
26{
27 VALIDATE_ARGUMENT_RANGE(paramT, 0, 1);
28
29 double w0 = startPoint.GetW();
30 XYZ P0 = XYZ(startPoint.GetWX()/w0, startPoint.GetWY()/w0, startPoint.GetWZ()/w0);
31 double w1 = middlePoint.GetW();
32 XYZ P1 = XYZ(middlePoint.GetWX() / w0, middlePoint.GetWY() / w0, middlePoint.GetWZ() / w0);
33 double w2 = endPoint.GetW();
34 XYZ P2 = XYZ(endPoint.GetWX() / w0, endPoint.GetWY() / w0, endPoint.GetWZ() / w0);
35
36 return ((1 - paramT) * (1 - paramT) * w0 * P0 + 2 * paramT * (1 - paramT) * w1 * P1 + paramT * paramT * w2 * P2) / ((1 - paramT) * (1 - paramT) * w0 + 2 * paramT * (1 - paramT) * w1 + paramT * paramT * w2);
37}

39bool LNLib::BezierCurve::ComputerMiddleControlPointsOnQuadraticCurve(const XYZ& startPoint, const XYZ& startTangent, const XYZ& endPoint, const XYZ& endTangent, std::vector<XYZW>& controlPoints)
40{
41 XYZ chord = (endPoint - startPoint).Normalize();
42 XYZ nST = const_cast<XYZ&>(startTangent).Normalize();
43 XYZ nET = const_cast<XYZ&>(endTangent).Normalize();
44
45 if (nST.IsAlmostEqualTo(chord) ||
46 nST.IsAlmostEqualTo(chord.Negative()))
47 {
48 if (nST.IsAlmostEqualTo(nET) ||
49 nST.IsAlmostEqualTo(nET.Negative()))
50 {
51 controlPoints.emplace_back((startPoint + endPoint) / 2, 1);
52 return true;
53 }
54 }
55
56 double alf1 = 0.0;
57 double alf2 = 0.0;
58
59 XYZ R(0, 0, 0);
60 double weight = 0.0;
61 auto type = Intersection::ComputeRays(startPoint, nST, endPoint, nET, alf1, alf2, R);
62 if (type == CurveCurveIntersectionType::Intersecting)
63 {
64 if (startPoint.IsAlmostEqualTo(R) || endPoint.IsAlmostEqualTo(R))
65 {
66 R = 0.5 * (startPoint + endPoint);
67 bool result = Interpolation::ComputerWeightForRationalQuadraticInterpolation(startPoint, R, endPoint, weight);
68 if (!result) return false;
69 controlPoints.emplace_back(XYZW(R, weight));
70 return true;
71 }
72
73 if (MathUtils::IsGreaterThan(alf1, 0.0) && MathUtils::IsLessThan(alf2, 0.0))
74 {
75 bool result = Interpolation::ComputerWeightForRationalQuadraticInterpolation(startPoint, R, endPoint, weight);
76 if (!result) return false;
77 controlPoints.emplace_back(XYZW(R, weight));
78 return true;
79 }
80 }
81
82 XYZ SE = endPoint - startPoint;
83 if (SE.Normalize().IsAlmostEqualTo(startTangent) && SE.Normalize().IsAlmostEqualTo(endTangent))
84 {
85 R = 0.5 * (startPoint + endPoint);
86 bool result = Interpolation::ComputerWeightForRationalQuadraticInterpolation(startPoint, R, endPoint, weight);
87 if (!result) return false;
88 controlPoints.emplace_back(XYZW(R, weight));
89 return true;
90 }
91
92 double gamma1 = 0.0;
93 double gamma2 = 0.0;
94
95 if (type != CurveCurveIntersectionType::Intersecting)
96 {
97 gamma1 = 0.5 * SE.Length();
98 gamma2 = gamma1;
99 }
100 else
101 {
102 XYZ SR = R - startPoint;
103 XYZ ER = R - endPoint;
104
105 double theta0 = SR.DotProduct(SE) / (SR.Length() * SE.Length());
106 double theta1 = ER.DotProduct(SE) / (ER.Length() * SE.Length());
107
108 double alpha = 2.0 / 3.0;
109 gamma1 = 0.5 * SE.Length() / (1.0 + alpha * theta1 + (1 - alpha) * theta0);
110 gamma2 = 0.5 * SE.Length() / (1.0 + alpha * theta0 + (1 - alpha) * theta1);
111 }
112
113 XYZ R1 = startPoint + gamma1 * startTangent;
114 XYZ R2 = endPoint - gamma2 * endTangent;
115 XYZ Qk = (gamma1 * R2 + gamma2 * R1) / (gamma1 + gamma2);
116
117 double weight1 = 0.0;
118 bool result = Interpolation::ComputerWeightForRationalQuadraticInterpolation(startPoint, R1, Qk, weight1);
119 if (!result) return false;
120
121 double weight2 = 0.0;
122 result = Interpolation::ComputerWeightForRationalQuadraticInterpolation(Qk, R2, endPoint, weight2);
123 if (!result) return false;
124
125 controlPoints.emplace_back(XYZW(R1, weight1));
126 controlPoints.emplace_back(XYZW(Qk, 1.0));
127 controlPoints.emplace_back(XYZW(R2, weight2));
128
129 return true;
130}
