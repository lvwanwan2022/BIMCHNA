22bool LNLib::Projection::PointToLine(const XYZ& start, const XYZ& end, const XYZ& point, XYZ& projectPoint)
23{
24 double length = start.Distance(end);
25 if (MathUtils::IsAlmostEqualTo(length, 0.0))
26 {
27 return false;
28 }
29 double param = (point - start).DotProduct(end - start) / (length * length);
30 if (MathUtils::IsGreaterThan(param, 1.0) || MathUtils::IsLessThan(param,0.0))
31 {
32 return false;
33 }
34 param = std::max(0.0, std::min(1.0, param));
35 projectPoint = start + param * (end - start);
36 return true;
37}

39LNLib::XYZ LNLib::Projection::Stereographic(const XYZ& pointOnSphere, double radius)
40{
41 double x = pointOnSphere.GetX();
42 double y = pointOnSphere.GetY();
43 double z = pointOnSphere.GetZ();
44
45 double alpha = -2 * radius / (z - 2 * radius);
46 double u = alpha * x;
47 double v = alpha * y;
48
49 return XYZ(u, v, 0);
50}
