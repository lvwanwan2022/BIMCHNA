97XYZ LNLib::XYZW::ToXYZ(bool divideWeight)const
98{
99 if (divideWeight)
100 {
101 double w = m_xyzw[3];
102 if (MathUtils::IsAlmostEqualTo(w, 0.0))
103 {
104 return XYZ(m_xyzw[0], m_xyzw[1], m_xyzw[2]);
105 }
106 else
107 {
108 return XYZ(m_xyzw[0] / w, m_xyzw[1] / w, m_xyzw[2] / w);
109 }
110 }
111 else
112 {
113 return XYZ(m_xyzw[0], m_xyzw[1], m_xyzw[2]);
114 }
115}

124double LNLib::XYZW::Distance(const XYZW& another) const
125{
126 double squareValue = pow((another.GetWX() - m_xyzw[0]), 2) + pow((another.GetWY() - m_xyzw[1]), 2) + pow((another.GetWZ() - m_xyzw[2]), 2) + pow((another.GetW() - m_xyzw[3]), 2);
127 return std::sqrt(squareValue);
