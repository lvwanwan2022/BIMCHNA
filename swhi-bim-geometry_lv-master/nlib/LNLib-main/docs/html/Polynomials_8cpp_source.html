22double Polynomials::Horner(int degree, const std::vector<double>& coefficients, double paramT)
23{
24 VALIDATE_ARGUMENT(degree > 0, "degree", "Degree must be greater than zero.");
25 VALIDATE_ARGUMENT(ValidationUtils::IsValidBezier(degree, coefficients.size()), "degree", "Coefficients size equal degree plus one.");
26 double result = coefficients[degree];
27 for (int i = degree - 1; i >= 0; i--)
28 {
29 result = result * paramT + coefficients[i];
30 }
31 return result;

34double Polynomials::Bernstein(int index, int degree, double paramT)
35{
36 if (index < 0 ||
37 index > degree)
38 {
39 return 0.0;
40 }
41 if (index == 0 ||
42 index == degree)
43 {
44 return 1.0;
45 }
46
47 VALIDATE_ARGUMENT(degree >= 0, "degree", "Degree must be greater than or equal zero.");
48 VALIDATE_ARGUMENT_RANGE(paramT, 0.0, 1.0);
49
50 std::vector<double> temp(degree + 1,0.0);
51 temp[degree - index] = 1.0;
52 double t1 = 1.0 - paramT;
53
54 for (int k = index; k <= degree; k++)
55 {
56 for (int j = degree; j >= k; j--)
57 {
58 temp[j] = t1 * temp[j] + paramT * temp[j - 1];
59 }
60 }
61 return temp[degree];
62}

64std::vector<double> Polynomials::AllBernstein(int degree, double paramT)
65{
66 VALIDATE_ARGUMENT(degree > 0, "degree", "Degree must be greater than zero.");
67 VALIDATE_ARGUMENT_RANGE(paramT, 0.0, 1.0);
68
69 std::vector<double> bernsteinArray(degree + 1);
70 bernsteinArray[0] = 1.0;
71
72 double t1 = 1.0 - paramT;
73 for (int j = 1; j <= degree; j++)
74 {
75 double saved = 0.0;
76 for (int k = 0; k < j; k++)
77 {
78 double temp = bernsteinArray[k];
79 bernsteinArray[k] = saved + t1 * temp;
80 saved = paramT * temp;
81 }
82 bernsteinArray[j] = saved;
83 }
84 return bernsteinArray;
85}

125int LNLib::Polynomials::GetKnotSpanIndex(int degree, const std::vector<double>& knotVector, double paramT)
126{
127 VALIDATE_ARGUMENT(degree >= 0, "degree", "Degree must be greater than or equal zero.");
128 VALIDATE_ARGUMENT(knotVector.size() > 0, "knotVector", "KnotVector size must be greater than zero.");
129 VALIDATE_ARGUMENT(ValidationUtils::IsValidKnotVector(knotVector), "knotVector", "KnotVector must be a nondecreasing sequence of real numbers.");
130 VALIDATE_ARGUMENT_RANGE(paramT, knotVector[0], knotVector[knotVector.size() - 1]);
131
132 int n = knotVector.size() - degree - 2;
133 if (MathUtils::IsGreaterThanOrEqual(paramT, knotVector[n + 1]))
134 {
135 return n;
136 }
137 if (MathUtils::IsLessThanOrEqual(paramT, knotVector[degree]))
138 {
139 return degree;
140 }
141
142 int low = degree;
143 int high = n + 1;
144 int mid = (low + high) / 2;
145
146 while (paramT < knotVector[mid] ||
147 paramT >= knotVector[mid + 1])
148 {
149 if (paramT < knotVector[mid])
150 {
151 high = mid;
152 }
153 else
154 {
155 low = mid;
156 }
157 mid = (low + high) / 2;
158 }
159 return mid;
160}

162void LNLib::Polynomials::BasisFunctions(int spanIndex, int degree, const std::vector<double>& knotVector, double paramT, double* basisFunctions)
163{
164 VALIDATE_ARGUMENT(spanIndex >= 0, "spanIndex", "SpanIndex must be greater than or equal zero.");
165 VALIDATE_ARGUMENT(degree >= 0 && degree <= Constants::NURBSMaxDegree, "degree", "Degree must be greater than or equal zero and not exceed the maximun degree.");
166 VALIDATE_ARGUMENT(knotVector.size() > 0, "knotVector", "KnotVector size must be greater than zero.");
167 VALIDATE_ARGUMENT(ValidationUtils::IsValidKnotVector(knotVector), "knotVector", "KnotVector must be a nondecreasing sequence of real numbers.");
168 VALIDATE_ARGUMENT_RANGE(paramT, knotVector[0], knotVector[knotVector.size() - 1]);
169
170 basisFunctions[0] = 1.0;
171
172 double left[Constants::NURBSMaxDegree + 1];
173 double right[Constants::NURBSMaxDegree + 1];
174
175 for (int j = 1; j <= degree; j++)
176 {
177 left[j] = paramT - knotVector[spanIndex + 1 - j];
178 right[j] = knotVector[spanIndex + j] - paramT;
179
180 double saved = 0.0;
181
182 for (int r = 0; r < j; r++)
183 {
184 double temp = basisFunctions[r] / (right[r + 1] + left[j - r]);
185 basisFunctions[r] = saved + right[r + 1] * temp;
186 saved = left[j - r] * temp;
187 }
188 basisFunctions[j] = saved;
189 }

193std::vector<std::vector<double>> LNLib::Polynomials::BasisFunctionsDerivatives(int spanIndex, int degree,
194 int derivative, const std::vector<double>& knotVector, double paramT)
195{
196 VALIDATE_ARGUMENT(spanIndex >= 0, "spanIndex", "SpanIndex must be greater than or equal zero.");
197 VALIDATE_ARGUMENT(degree >= 0 && degree <= Constants::NURBSMaxDegree, "degree", "Degree must be greater than or equal zero and not exceed the maximun degree.");
198 VALIDATE_ARGUMENT(derivative <= degree, "derivative", "Derivative must not be greater than degree.");
199 VALIDATE_ARGUMENT(knotVector.size() > 0, "knotVector", "KnotVector size must be greater than zero.");
200 VALIDATE_ARGUMENT(ValidationUtils::IsValidKnotVector(knotVector), "knotVector", "KnotVector must be a nondecreasing sequence of real numbers.");
201 VALIDATE_ARGUMENT_RANGE(paramT, knotVector[0], knotVector[knotVector.size() - 1]);
202
203 std::vector<std::vector<double>> derivatives(derivative + 1, std::vector<double>(degree + 1));
204 std::vector<std::vector<double>> ndu(degree + 1,std::vector<double>(degree + 1));
205
206 ndu[0][0] = 1.0;
207
208 double left[Constants::NURBSMaxDegree + 1];
209 double right[Constants::NURBSMaxDegree + 1];
210
211 double saved = 0.0;
212 double temp = 0.0;
213
214 for (int j = 1; j <= degree; j++)
215 {
216 left[j] = paramT - knotVector[spanIndex + 1 - j];
217 right[j] = knotVector[spanIndex + j] - paramT;
218
219 saved = 0.0;
220 for (int r = 0; r < j; r++)
221 {
222 ndu[j][r] = right[r + 1] + left[j - r];
223 temp = ndu[r][j - 1] / ndu[j][r];
224
225 ndu[r][j] = saved + right[r + 1] * temp;
226 saved = left[j - r] * temp;
227 }
228 ndu[j][j] = saved;
229 }
230
231 for (int j = 0; j <= degree; j++)
232 {
233 derivatives[0][j] = ndu[j][degree];
234 }
235
236 std::vector<std::vector<double>> a(2,std::vector<double>(degree + 1));
237 for (int r = 0; r <= degree; r++)
238 {
239 int s1 = 0;
240 int s2 = 1;
241 a[0][0] = 1.0;
242
243 for (int k = 1; k <= derivative; k++)
244 {
245 double d = 0.0;
246 int rk = r - k;
247 int pk = degree - k;
248
249 if (r >= k)
250 {
251 a[s2][0] = a[s1][0] / ndu[pk + 1][rk];
252 d = a[s2][0] * ndu[rk][pk];
253 }
254
255 int j1 = 0;
256 int j2 = 0;
257
258 if (rk >= -1)
259 j1 = 1;
260 else
261 j1 = -rk;
262
263 if (r - 1 <= pk)
264 j2 = k - 1;
265 else
266 j2 = degree - r;
267
268 for (int j = j1; j <= j2; j++)
269 {
270 a[s2][j] = (a[s1][j] - a[s1][j - 1]) / ndu[pk + 1][rk + j];
271 d += a[s2][j] * ndu[rk + j][pk];
272 }
273 if (r <= pk)
274 {
275 a[s2][k] = -a[s1][k - 1] / ndu[pk + 1][r];
276 d += a[s2][k] * ndu[r][pk];
277 }
278 derivatives[k][r] = d;
279
280 int temp = s1;
281 s1 = s2;
282 s2 = temp;
283 }
284 }
285
286 int r = degree;
287 for (int k = 1; k <= derivative; k++)
288 {
289 for (int j = 0; j <= degree; j++)
290 {
291 derivatives[k][j] *= r;
292 }
293 r *= degree - k;
294 }
295 return derivatives;
296}

298void LNLib::Polynomials::BasisFunctionsFirstOrderDerivative(int spanIndex, int degree, const std::vector<double>& knotVector, double paramT, double derivatives[2][Constants::NURBSMaxDegree + 1])
299{
300 VALIDATE_ARGUMENT(spanIndex >= 0, "spanIndex", "SpanIndex must be greater than or equal zero.");
301 VALIDATE_ARGUMENT(degree >= 0 && degree <= Constants::NURBSMaxDegree, "degree", "Degree must be greater than or equal zero and not exceed the maximun degree.");
302 VALIDATE_ARGUMENT(1 <= degree, "derivative", "Derivative must not be greater than degree.");
303 VALIDATE_ARGUMENT(knotVector.size() > 0, "knotVector", "KnotVector size must be greater than zero.");
304 VALIDATE_ARGUMENT(ValidationUtils::IsValidKnotVector(knotVector), "knotVector", "KnotVector must be a nondecreasing sequence of real numbers.");
305 VALIDATE_ARGUMENT_RANGE(paramT, knotVector[0], knotVector[knotVector.size() - 1]);
306
307 double ndu[Constants::NURBSMaxDegree + 1][Constants::NURBSMaxDegree + 1];
308
309 ndu[0][0] = 1.0;
310
311 double left[Constants::NURBSMaxDegree + 1];
312 double right[Constants::NURBSMaxDegree + 1];
313
314 double saved = 0.0;
315 double temp = 0.0;
316
317 for (int j = 1; j <= degree; j++)
318 {
319 left[j] = paramT - knotVector[spanIndex + 1 - j];
320 right[j] = knotVector[spanIndex + j] - paramT;
321
322 saved = 0.0;
323 for (int r = 0; r < j; r++)
324 {
325 ndu[j][r] = right[r + 1] + left[j - r];
326 temp = ndu[r][j - 1] / ndu[j][r];
327
328 ndu[r][j] = saved + right[r + 1] * temp;
329 saved = left[j - r] * temp;
330 }
331 ndu[j][j] = saved;
332 }
333
334 for (int j = 0; j <= degree; j++)
335 {
336 derivatives[0][j] = ndu[j][degree];
337 }
338
339 double a[2][Constants::NURBSMaxDegree + 1];
340 for (int r = 0; r <= degree; r++)
341 {
342 int s1 = 0;
343 int s2 = 1;
344 a[0][0] = 1.0;
345
346 const int k = 1;
347 {
348 double d = 0.0;
349 int rk = r - k;
350 int pk = degree - k;
351
352 if (r >= k)
353 {
354 a[s2][0] = a[s1][0] / ndu[pk + 1][rk];
355 d = a[s2][0] * ndu[rk][pk];
356 }
357
358 int j1 = 0;
359 int j2 = 0;
360
361 if (rk >= -1)
362 j1 = 1;
363 else
364 j1 = -rk;
365
366 if (r - 1 <= pk)
367 j2 = k - 1;
368 else
369 j2 = degree - r;
370
371 for (int j = j1; j <= j2; j++)
372 {
373 a[s2][j] = (a[s1][j] - a[s1][j - 1]) / ndu[pk + 1][rk + j];
374 d += a[s2][j] * ndu[rk + j][pk];
375 }
376 if (r <= pk)
377 {
378 a[s2][k] = -a[s1][k - 1] / ndu[pk + 1][r];
379 d += a[s2][k] * ndu[r][pk];
380 }
381 derivatives[k][r] = d;
382
383 int temp = s1;
384 s1 = s2;
385 s2 = temp;
386 }
387 }
388
389 int r = degree;
390
391 for (int j = 0; j <= degree; j++)
392 {
393 derivatives[1][j] *= r;
394 }
395}

397double LNLib::Polynomials::OneBasisFunction(int spanIndex, int degree, const std::vector<double>& knotVector, double paramT)
398{
399 VALIDATE_ARGUMENT(spanIndex >= 0, "spanIndex", "SpanIndex must be greater than or equal zero.");
400 VALIDATE_ARGUMENT(degree > 0, "degree", "Degree must be greater than zero.");
401 VALIDATE_ARGUMENT(knotVector.size() > 0, "knotVector", "KnotVector size must be greater than zero.");
402 VALIDATE_ARGUMENT(ValidationUtils::IsValidKnotVector(knotVector), "knotVector", "KnotVector must be a nondecreasing sequence of real numbers.");
403 VALIDATE_ARGUMENT_RANGE(paramT, knotVector[0], knotVector[knotVector.size() - 1]);
404
405 int m = knotVector.size() - 1;
406 if ((spanIndex == 0 && MathUtils::IsAlmostEqualTo(paramT, knotVector[0])) ||
407 (spanIndex == m - degree - 1 && MathUtils::IsAlmostEqualTo(paramT, knotVector[m])))
408 {
409 return 1.0;
410 }
411 if (MathUtils::IsLessThan(paramT, knotVector[spanIndex]) ||
412 MathUtils::IsGreaterThanOrEqual(paramT, knotVector[spanIndex + degree + 1]))
413 {
414 return 0.0;
415 }
416
417 std::vector<double> N(degree + spanIndex + 1, 0.0);
418 for (int j = 0; j <= degree; j++)
419 {
420 if (MathUtils::IsGreaterThanOrEqual(paramT, knotVector[spanIndex + j]) &&
421 MathUtils::IsLessThan(paramT, knotVector[spanIndex + j + 1]))
422 {
423 N[j] = 1.0;
424 }
425 }
426
427 for (int k = 1; k <= degree; k++)
428 {
429 double saved = 0.0;
430 if (!MathUtils::IsAlmostEqualTo(N[0], 0.0))
431 {
432 saved = ((paramT - knotVector[spanIndex]) * N[0]) / (knotVector[spanIndex + k] - knotVector[spanIndex]);
433 }
434
435 for (int j = 0; j < degree - k + 1; j++)
436 {
437 double knotLeft = knotVector[spanIndex + j + 1];
438 double knotRight = knotVector[spanIndex + j + k + 1];
439 if (MathUtils::IsAlmostEqualTo(N[j + 1], 0.0))
440 {
441 N[j] = saved;
442 saved = 0.0;
443 }
444 else
445 {
446 double temp = N[j + 1] / (knotRight - knotLeft);
447 N[j] = saved + (knotRight - paramT) * temp;
448 saved = (paramT - knotLeft) * temp;
449 }
450 }
451 }
452 return N[0];
453}

455std::vector<double> LNLib::Polynomials::OneBasisFunctionDerivative(int spanIndex, int degree, int derivative, const std::vector<double>& knotVector, double paramT)
456{
457 VALIDATE_ARGUMENT(spanIndex >= 0, "spanIndex", "SpanIndex must be greater than or equal zero.");
458 VALIDATE_ARGUMENT(degree > 0, "degree", "Degree must be greater than zero.");
459 VALIDATE_ARGUMENT(derivative <= degree, "derivative", "Derivative must not be greater than degree.");
460 VALIDATE_ARGUMENT(knotVector.size() > 0, "knotVector", "KnotVector size must be greater than zero.");
461 VALIDATE_ARGUMENT(ValidationUtils::IsValidKnotVector(knotVector), "knotVector", "KnotVector must be a nondecreasing sequence of real numbers.");
462 VALIDATE_ARGUMENT_RANGE(paramT, knotVector[0], knotVector[knotVector.size() - 1]);
463
464 std::vector<double> derivatives(derivative + 1, 0.0);
465
466 if (MathUtils::IsLessThan(paramT, knotVector[spanIndex]) ||
467 MathUtils::IsGreaterThanOrEqual(paramT, knotVector[spanIndex + degree + 1]))
468 {
469 return derivatives;
470 }
471
472 std::vector<std::vector<double>> N(degree + 1, std::vector<double>(degree + 1,0.0));
473
474 for (int j = 0; j <= degree; j++)
475 {
476 if (MathUtils::IsGreaterThanOrEqual(paramT, knotVector[spanIndex + j]) &&
477 MathUtils::IsLessThan(paramT, knotVector[spanIndex + j + 1]))
478 {
479 N[j][0] = 1.0;
480 }
481 }
482 for (int k = 1; k <= degree; k++)
483 {
484 double saved = 0.0;
485 if (!MathUtils::IsAlmostEqualTo(N[0][k - 1], 0.0))
486 {
487 saved = ((paramT - knotVector[spanIndex]) * N[0][k - 1]) / (knotVector[spanIndex + k] - knotVector[spanIndex]);
488 }
489 for (int j = 0; j < degree - k + 1; j++)
490 {
491 double knotLeft = knotVector[spanIndex + j + 1];
492 double knotRight = knotVector[spanIndex + j + k + 1];
493
494 if (MathUtils::IsAlmostEqualTo(N[j + 1][k - 1], 0.0))
495 {
496 N[j][k] = saved;
497 saved = 0.0;
498 }
499 else
500 {
501 double temp = N[j + 1][k - 1] / (knotRight - knotLeft);
502 N[j][k] = saved + (knotRight - paramT) * temp;
503 saved = (paramT - knotLeft) * temp;
504 }
505 }
506 }
507
508 derivatives[0] = N[0][degree];
509
510 for (int k = 1; k <= derivative; k++)
511 {
512 std::vector<double> ND(k+1,0.0);
513 for (int j = 0; j <= k; j++)
514 {
515 ND[j] = N[j][degree - k];
516 }
517 for (int jj = 1; jj <= k; jj++)
518 {
519 double saved = MathUtils::IsAlmostEqualTo(ND[0], 0.0)?
520 0.0: ND[0] / (knotVector[spanIndex + degree - k + jj] - knotVector[spanIndex]);
521 for (int j = 0; j < k - jj + 1; j++)
522 {
523 double knotLeft = knotVector[spanIndex + j + 1];
524 double knotRight = knotVector[spanIndex + j + degree - k + jj + 1];
525
526 if (MathUtils::IsAlmostEqualTo(ND[j + 1], 0.0))
527 {
528 ND[j] = (degree - k + jj) * saved;
529 saved = 0.0;
530 }
531 else
532 {
533 double temp = ND[j + 1] / (knotRight - knotLeft);
534 ND[j] = (degree - k + jj) * (saved - temp);
535 saved = temp;
536 }
537 }
538 }
539 derivatives[k] = ND[0];
540 }
541 return derivatives;
542}

544std::vector<std::vector<double>> LNLib::Polynomials::AllBasisFunctions(int spanIndex, int degree, const std::vector<double>& knotVector, double knot)
545{
546 VALIDATE_ARGUMENT(spanIndex >= 0, "spanIndex", "SpanIndex must be greater than or equal zero.");
547 VALIDATE_ARGUMENT(degree >= 0 && degree <= Constants::NURBSMaxDegree, "degree", "Degree must be greater than or equal zero and not exceed the maximun degree.");
548 VALIDATE_ARGUMENT(knotVector.size() > 0, "knotVector", "KnotVector size must be greater than zero.");
549 VALIDATE_ARGUMENT(ValidationUtils::IsValidKnotVector(knotVector), "knotVector", "KnotVector must be a nondecreasing sequence of real numbers.");
550 VALIDATE_ARGUMENT_RANGE(knot, knotVector[0], knotVector[knotVector.size() - 1]);
551
552 std::vector<std::vector<double>> result(degree + 1, std::vector<double>(degree + 1));
553 for (int i = 0; i <= degree; i++)
554 {

557 for (int j = 0; j <= i; j++)
558 {
559 result[j][i] = basis[j];
560 }
561 }
562 return result;
563}

565std::vector<std::vector<double>> LNLib::Polynomials::BezierToPowerMatrix(int degree)
566{
567 std::vector<std::vector<double>> matrix(degree+1, std::vector<double>(degree + 1));
568 for (int i = 0; i < degree; i++)
569 {
570 for (int j = i + 1; j <= degree; j++)
571 {
572 matrix[i][j] = 0.0;
573 }
574 }
575
576 matrix[0][0] = matrix[degree][degree] = 1.0;
577 matrix[degree][0] = degree % 2 == 0 ? -1.0 : 1.0;
578
579 double sign = -1.0;
580 for (int i = 1; i < degree; i++)
581 {
582 matrix[i][i] = MathUtils::Binomial(degree,i);
583 matrix[i][0] = matrix[degree][degree - 1] = sign * matrix[i][i];
584 sign = -sign;
585 }
586
587 int k1 = (degree + 1) / 2;
588 int pk = degree - 1;
589 for (int k = 1; k < k1; k++)
590 {
591 sign = -1.0;
592 for (int j = k + 1; j <= pk; j++)
593 {
594 matrix[j][k] = matrix[pk][degree - j] = sign * MathUtils::Binomial(degree, k) * MathUtils::Binomial(degree - k, j - k);
595 sign = -sign;
596 }
597 pk = pk - 1;
598 }
599 return matrix;
600}

602std::vector<std::vector<double>> LNLib::Polynomials::PowerToBezierMatrix(int degree, const std::vector<std::vector<double>>& matrix)
603{
604 std::vector<std::vector<double>> inverseMatrix(degree + 1, std::vector<double>(degree + 1));
605 for (int i = 0; i < degree; i++)
606 {
607 for (int j = i + 1; j <= degree; j++)
608 {
609 inverseMatrix[i][j] = 0.0;
610 }
611 }
612
613 for (int i = 0; i <= degree; i++)
614 {
615 inverseMatrix[i][0] = inverseMatrix[degree][i] = 1.0;
616 inverseMatrix[i][i] = 1.0 / (matrix[i][i]);
617 }
618
619 int k1 = (degree + 1) / 2;
620 int pk = degree - 1;
621
622 for (int k = 1; k < k1; k++)
623 {
624 for (int j = k + 1; j < pk; j++)
625 {
626 double d = 0.0;
627 for (int i = k; i < j; i++)
628 {
629 d = d - matrix[j][i] * inverseMatrix[i][k];
630 }
631 inverseMatrix[j][k] = d / (inverseMatrix[j][j]);
632 inverseMatrix[pk][degree - j] = inverseMatrix[j][k];
633 }
634 pk = pk - 1;
635 }
636 return inverseMatrix;
