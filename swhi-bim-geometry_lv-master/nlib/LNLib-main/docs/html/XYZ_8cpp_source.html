112double LNLib::XYZ::AngleTo(const XYZ& another) const
113{
114 XYZ ntTemp = const_cast<XYZ&>(*this).Normalize();
115 XYZ naTemp = const_cast<XYZ&>(another).Normalize();
116
117 if (ntTemp.IsZero() ||
118 naTemp.IsZero())
119 {
120 return 0.0;
121 }
122 double dot = ntTemp.DotProduct(naTemp);
123 dot = dot < -1.0 ? -1.0 : dot > 1.0 ? 1.0 : dot;
124 return acos(dot);
125}

174XYZ LNLib::XYZ::CreateRandomOrthogonal(const XYZ& xyz)
175{
176 XYZ current = xyz;
177 XYZ normal = current.Normalize();
178 XYZ tangent = normal.CrossProduct(XYZ(-normal.GetZ(), normal.GetX(), normal.GetY()));
179 XYZ bitangent = normal.CrossProduct(tangent);
180
181 std::random_device rd;
182 std::mt19937 gen(rd());
183 std::uniform_int_distribution<> distrib(-Constants::Pi, Constants::Pi);
184 double angle = distrib(gen);
185 return (tangent * std::sin(angle) + bitangent * std::cos(angle)).Normalize();
186}
